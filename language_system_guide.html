Add to Other Pages

To add language support to any page:

<?php
// 1. Include at top of file
require_once 'includes/language.php';

// 2. Set HTML lang and dir
<html lang="<?= $current_lang ?>" dir="<?= isRTL() ? 'rtl' : 'ltr' ?>">

// 3. Add language switcher to navbar
<?php include 'includes/language_switcher.php'; ?>

// 4. Use translations
<h1><?= t('welcome') ?></h1>
?>
